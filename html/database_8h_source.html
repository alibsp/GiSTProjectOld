176 class FixedSizeAllocator
177 {
178  struct Hole {
179  Hole* next;
180  offs_t offs;
181  };
182 
183  size_t minSize;
184  size_t maxSize;
185  size_t quantum;
186  size_t nChains;
187  size_t bufSize;
188  Hole** chains;
189  Hole* vacant;
190  Hole* holes;
191 
192  public:
193  size_t hits;
194  size_t faults;
195  size_t retries;
196 
197  FixedSizeAllocator();
198  ~FixedSizeAllocator();
199 
200  void init(size_t minSize, size_t maxSize, size_t quantum, size_t bufSize);
201  void reset();
202 
203  offs_t allocate(size_t size) {
204  if (size - minSize <= maxSize - minSize) {
205  size_t i = (size - minSize + quantum - 1) / quantum;
206  Hole* hole = chains[i];
207  if (hole != NULL) {
208  hits += 1;
209  chains[i] = hole->next;
210  hole->next = vacant;
211  vacant = hole;
212  return hole->offs;
213  }
214  faults += 1;
215  }
216  return 0;
217  }
218 
219  bool deallocate(offs_t offs, size_t size) {
220  if (vacant != NULL && size - minSize <= maxSize - minSize) {
221  size_t i = (size - minSize + quantum - 1) / quantum;
222  Hole* hole = vacant;
223  vacant = hole->next;
224  hole->next = chains[i];
225  chains[i] = hole;
226  hole->offs = offs;
227  return true;
228  }
229  return false;
230  }
231 };
